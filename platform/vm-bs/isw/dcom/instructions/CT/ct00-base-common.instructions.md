##  HELPER FUNCTIONS

### **Security Access Key Calculation**

##  HELPER FUNCTIONS

### **Security Access Key Calculation**

```c
static uint8 key[8] = {0};

static void calculate_key(uint8 *seed, uint8 SecLevel_u8)
{
    switch (SecLevel_u8)
    {
        case 1:  /* EOL Level */
            key[1] = seed[0];
            key[0] = seed[1];
            break;
        case 2:  /* Service Level */
            key[0] = seed[0] ^ seed[1];
            key[1] = seed[0];
            break;
        case 3:  /* Fail condition for negative testing */
            key[0] = 0x0F & seed[0];
            key[1] = 0x0F & seed[1];
            break;
        default:
            break;
    }
}
```

### **Security Level Constants (Optional)**

```c
#define C_SECURITY_LEVEL_L1  0x01
#define C_SECURITY_LEVEL_L2  0x02
```

##  TEST CASE STRUCTURE

### **Basic Test Function Template**

```c
/*
  @testlists <TestCategory>
  <Test case description>
*/
SWTEST void TC_<Service>_<DID/PID/RID>_<SequenceNumber>(void)
{
    /* Variable declarations */
    uint8 seed[8] = {0};
  
    /* MESG definitions (if needed) */
    RBMESG_DefineMESGDef(<MessageName>);
  
    /* System initialization */
    SWT_RunSystemMS(10000);
  
    /* Session management */
    [Session setup]
  
    /* Security unlock (if required) */
    [Security access]
  
    /* Test setup - Set signal values */
    [Signal configuration]
  
    /* Test execution */
    [Diagnostic requests and validations]
  
    /* Cleanup (if needed) */
}
```

### **Session Management**

**Default Session:**

```c
//Start: Default_Diagnostic_Session
SWT_RunSystemMS(10000);
SWT_Eval(HSWFT_DiagSendRequest("UDS", "10 01", "Default Session"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "50 01 00 32 01 F4", 6));
SWT_RunSystemMS(250);
SWT_Eval(HSWFT_DiagSendRequest("UDS", "22 F1 86", "Read Active Session"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "62 F1 86 01", 4));
SWT_RunSystemMS(3750);
//End  : Default_Diagnostic_Session
```

**Extended Session:**

```c
//Start: Extended_Diagnostic_Session
SWT_RunSystemMS(10000);
SWT_Eval(HSWFT_DiagSendRequest("UDS", "10 01", "Default Session"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "50 01 00 32 01 F4", 6));
SWT_Eval(HSWFT_DiagSendRequest("UDS", "22 F1 86", "Read Active Session"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "62 F1 86 01", 4));
SWT_RunSystemMS(250);
SWT_Eval(HSWFT_DiagSendRequest("UDS", "10 03", "Extended Session"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "50 03 00 32 01 F4", 6));
SWT_RunSystemMS(3750);
//End  : Extended_Diagnostic_Session
```

**Shorter Setup:**

```c
SWT_RunSystemMS(10000);
SWT_Eval(HSWFT_DiagSendRequest("UDS", "10 01", "Default Session"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "50 01 00 32 01 F4", 6));
SWT_Eval(HSWFT_DiagSendRequest("UDS", "10 03", "Extended Session"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "50 03 00 32 01 F4", 6));
HSWFT_DiagDelayMS("UDS", 3750);
```

### **Security Access**

**EOL Level (Level 1):**

```c
//Start: Unlock_SecurityLevel_EOL
SWT_Eval(HSWFT_DiagSendRequest("UDS", "27 01", "Request Seed"));
SWT_RunSystemMS(1000);
SWT_Eval(HSWFT_DiagReadSecurityResponse("UDS", "67 01", seed, 2));
calculate_key(seed, 1);
SWT_RunSystemMS(1000);
SWT_Eval(HSWFT_DiagSendSecurityUnlock("UDS", "27 02", key, 2, "Unlock"));
SWT_RunSystemMS(1500);
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "67 02", 2));
SWT_RunSystemMS(100);
//End  : Unlock_SecurityLevel_EOL
```

**Service Level (Level 2):**

```c
//Start: Unlock_SecurityLevel_Service
SWT_Eval(HSWFT_DiagSendRequest("UDS", "27 03", "Request Seed"));
SWT_RunSystemMS(1000);
SWT_Eval(HSWFT_DiagReadSecurityResponse("UDS", "67 03", seed, 2));
calculate_key(seed, 2);
SWT_RunSystemMS(1000);
SWT_Eval(HSWFT_DiagSendSecurityUnlock("UDS", "27 04", key, 2, "Unlock"));
SWT_RunSystemMS(1500);
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "67 04", 2));
SWT_RunSystemMS(100);
//End  : Unlock_SecurityLevel_Service
```

### **Signal Setup**

**Single Signal:**

```c
RBMESG_DefineMESGDef(<SignalName>);

l_<SignalName>.Qualifier = 0x00;  /* Valid */
l_<SignalName>.<Field> = <Value>;

RBMESG_SendMESGDef(<SignalName>);

SWT_RunSystemMS(10000);  /* Stabilization */
```

**Multiple Signals:**

```c
RBMESG_DefineMESGDef(Signal1);
RBMESG_DefineMESGDef(Signal2);
RBMESG_DefineMESGDef(Signal3);

l_Signal1.Qualifier = 0x00;
l_Signal1.Value = 0x1234;
l_Signal2.Qualifier = 0x00;
l_Signal2.Value = 0x5678;
l_Signal3.Qualifier = 0x00;
l_Signal3.Value = 0x9ABC;

RBMESG_SendMESGDef(Signal1);
RBMESG_SendMESGDef(Signal2);
RBMESG_SendMESGDef(Signal3);

SWT_RunSystemMS(10000);
```

**Mixed MESG Types:**

```c
/* Standard MESG */
RBMESG_DefineMESGDef(RBSSPWSS_WheelSpeed_FL);

/* Legacy MESG */
DefineMESGDef(RBAPLHELP_DiagnosisActuationRequestActive);

l_RBSSPWSS_WheelSpeed_FL.WheelSpeed = 0x1900;

RBMESG_SendMESGDef(RBSSPWSS_WheelSpeed_FL);
SendMESG(RBAPLHELP_DiagnosisActuationRequestActive, 1);

SWT_RunSystemMS(10000);
```

### **Timing Patterns**

```c
/* System stabilization */
SWT_RunSystemMS(10000);

/* Short delay */
SWT_RunSystemMS(250);

/* Medium delay */
SWT_RunSystemMS(1000);

/* Session timeout test */
SWT_RunSystemMS(3750);  /* Just before 5-second timeout */
SWT_RunSystemMS(6000);  /* After timeout */

/* Diagnostic delay */
HSWFT_DiagDelayMS("UDS", 3750);
HSWFT_DiagDelayMS("UDS", 5010);

/* Software reset */
SWT_SoftwareReset();
SWT_RunSystemMS(500);
```

### **DTC Management**

```c
/* Report DTC */
Dem_ReportErrorStatus(DemConf_DemEventParameter_<EventName>, DEM_EVENT_STATUS_FAILED);
SWT_RunSystemMS(500);

/* Clear DTC */
SWT_Eval(HSWFT_DiagSendRequest("UDS", "14 FF FF FF", "Clear all DTC"));
SWT_RunSystemMS(10000);  /* NvM write time */
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "54", 1));

/* Read DTC status */
SWT_Eval(HSWFT_DiagSendRequest("UDS", "19 01 05", "Retrieve active DTC count"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "59 01 FF 01 00 00", 6));

SWT_Eval(HSWFT_DiagSendRequest("UDS", "19 02 FF", "Retrieve active DTCs"));
SWT_Eval(HSWFT_DiagEvalResponse("UDS", "59 02 FF <StatusMask> <DTCs...>", <Length>));
```

##  TEST DESIGN CHECKLIST

### **Coverage Requirements**
- [ ] Positive scenarios (happy path)
- [ ] Negative scenarios (all applicable NRCs)
- [ ] Boundary conditions (min/max values)
- [ ] Edge cases (invalid, not available)
- [ ] Session management
- [ ] Timing requirements

### **Code Quality**
- [ ] Consistent indentation
- [ ] Clear comments for complex logic
- [ ] Section markers for test phases
- [ ] Meaningful test descriptions
- [ ] Requirements traceability

### **Test Independence**
- [ ] Self-contained test cases
- [ ] Initialize all required state
- [ ] Don't rely on execution order
- [ ] Cleanup handled properly

### **Timing & Synchronization**
- [ ] System initialization (10000ms)
- [ ] Between commands (250ms)
- [ ] Before session timeout (3750ms)
- [ ] NvM operations (10000ms)

### **Signal Handling**
- [ ] Define MESG before use
- [ ] Set qualifier appropriately
- [ ] Send signals before reading
- [ ] Allow stabilization time

##  COMMON MISTAKES

**Timing:** Insufficient delays, wrong timeout values, missing stabilization
**Signals:** Missing MESG definitions, wrong qualifier values, no stabilization delay
**Sessions:** Missing session setup, incorrect timeout handling
**Security:** Wrong key calculation, missing seed/key sequence
**NRCs:** Incomplete error testing, wrong NRC codes, missing edge cases
**Documentation:** Missing test descriptions, no requirements traceability

##  QUICK REFERENCE

### **Diagnostic Channels**

```c
"UDS"       - Standard UDS (physical)
"OBD"       - OBD (physical)
"OBD_FUNC"  - OBD (functional)
```

### **Common NRC Codes**

| NRC  | Name                                   | Description            |
| ---- | -------------------------------------- | ---------------------- |
| 0x10 | generalReject                          | General reject         |
| 0x11 | serviceNotSupported                    | Service not supported  |
| 0x12 | subfunctionNotSupported                | Subfunction not found  |
| 0x13 | incorrectMessageLengthOrInvalidFormat  | Length/format error    |
| 0x22 | conditionsNotCorrect                   | Conditions not correct |
| 0x24 | requestSequenceError                   | Wrong sequence         |
| 0x31 | requestOutOfRange                      | Out of range           |
| 0x33 | securityAccessDenied                   | Security locked        |
| 0x35 | invalidKey                             | Wrong key              |
| 0x36 | exceedNumberOfAttempts                 | Too many attempts      |
| 0x37 | requiredTimeDelayNotExpired            | Wait time not expired  |
| 0x7E | subfunctionNotSupportedInActiveSession | Blocked in session     |
| 0x7F | serviceNotSupportedInActiveSession     | Service blocked        |

### **Standard UDS Services**

| Service | Name                           | Description         |
| ------- | ------------------------------ | ------------------- |
| 0x10    | DiagnosticSessionControl       | Change session      |
| 0x11    | ECUReset                       | Reset ECU           |
| 0x14    | ClearDiagnosticInformation     | Clear DTCs          |
| 0x19    | ReadDTCInformation             | Read DTC status     |
| 0x22    | ReadDataByIdentifier           | Read data by DID    |
| 0x27    | SecurityAccess                 | Unlock security     |
| 0x28    | CommunicationControl           | Control comms       |
| 0x2E    | WriteDataByIdentifier          | Write data by DID   |
| 0x2F    | InputOutputControlByIdentifier | Control I/O         |
| 0x31    | RoutineControl                 | Execute routine     |
| 0x3E    | TesterPresent                  | Keep session alive  |
| 0x85    | ControlDTCSetting              | Enable/disable DTC  |

### **Typical Timing Values**

```c
10000 ms  - Initial system stabilization
50 ms     - P2 timeout (standard)
5000 ms   - P2* extended / Session timeout (S3)
250 ms    - After session change
1000 ms   - Between seed/key
500 ms    - After DTC report
3750 ms   - Before timeout test
6000 ms   - After timeout occurred
```
